关于print的操作

函数指针

int (*f)(int)

void(*f)()注意，*函数一定要带括号，不然就认为是一种声明返回值为void *的函数定义了。

加上括号就是代表这个是一个函数指针，而不是函数声明。参数列表就跟在函数名字后面。

但是其实函数名字本身就是一个引用，所以赋值的时候，让(*func)=fu就代表函数func指向fu那一块的内容。

函数名就相当于函数的入口地址，和 int*x=&a一样。只不过函数的时候函数名就是他的地址。
声明和使用的时候都记得要加括号。

两种使用方式：(*f)();和f()，二者是相同的。

有效的是函数名是定死的，但是函数指针可以变化。函数指针可以作为参数传递给其他参数，或者函数名作为参数传递。

可将函数指针作为参数，然后进来的时候
(void (*f)(),,int a, int b)

直接使用该形式函数指针：(*f)();

本质上函数的调用是执行了一个代码跳转，先把参数放到寄存器里面，然后跳转向这个函数的地址。

关于typedef

typedef void(*func)(void)

假如没有typedef的话，那么仅仅就是声明了一个函数指针，func是指向无参数无返回值的函数的第一个函数指针。

但是加上之后就是定义了一个 void(*)()的函数指针类型，相当于创造了一个新的类型（这个创造新类型也是typedef的用法）

typedef本义和define：define在预处理的时候被替换，然后 typefef是在编译的时候被解析

按照类似define的用法，不是 typedef void(*)() func这样好吗？

但是事实上void(*)()不是一个类型的，typedef void(*func)(void)定义函数的时候就是直接相当于定义了函数的类型。

/* --------------- local help functions --------------------- */
void print_char(fmt_callback_t out, void *data, char c, int length, int ladjust) {
    int i;
    输出一个字符。

    if (length < 1) {
        length = 1;
    }
    如果设定的长度小于1的话强制置1
    const char space = ' ';（一个新奇的用法，这个char的地址就相当于传进来一个字符串指针
    if (ladjust) {
        out(data, &c, 1); 注意要传指针
        for (i = 1; i < length; i++) {
            out(data, &space, 1);
        }
    } else {
        for (i = 0; i < length - 1; i++) {
            out(data, &space, 1);
        }
        out(data, &c, 1);
    }
}

void print_str(fmt_callback_t out, void *data, const char *s, int length, int ladjust) {
    一个输出字符串的函数。
    int i;
    int len = 0;
    const char *s1 = s;一个可变的，不改变形参的
    while (*s1++) {不是'\0'就++，这个len指的是这个字符串真实的长度。
        len++;
    }
    if (length < len) {
        length = len;length指的是设定的需要输出的对齐长度
    }

    if (ladjust) { 如果是左对齐，那么输出data，字符s，长度
        out(data, s, len);
        for (i = len; i < length; i++) {
            out(data, " ", 1);左对齐补充空白
        }
    } else {
        for (i = 0; i < length - len; i++) {
            out(data, " ", 1);这个的意思是输出一个空格，然后相当于用来右对齐补充空白。
        }
        out(data, s, len);
    }
}

void print_num(fmt_callback_t out, void *data, unsigned long u, int base, int neg_flag, int length,
           int ladjust, char padc, int upcase) {
            negflag代表存进来的是负数。
            base是进制基数
            length是要求的长度
            ladjust左对齐参数
            padc代表填充的字符。
            upcase指的是大小写状态
    * algorithm :
     *  1. prints the number from left to right in reverse form.
     算法：逆向从左到右输出数字
     *  2. fill the remaining spaces with padc if length is longer than
     *     the actual length
     如果要求大于实际长度的话补充实际空格
     *     TRICKY : if left adjusted, no "0" padding.
     如果左对齐，不要补充0
     *		    if negtive, insert  "0" padding between "0" and number.
    如果是负数的话在0和数字之间补充0
     *  3. if (!ladjust) we reverse the whole string including paddings
     *  4. otherwise we only reverse the actual string representing the num.
     */
        输出数字
    int actualLength = 0;实际的长度
    char buf[length + 70];将要输出的格式化字符串
    char *p = buf;可移动的字符指针
    int i;
    u是实际传进来的数字，
    do {
        int tmp = u % base;
        if (tmp <= 9) {
            *p++ = '0' + tmp;
        } else if (upcase) {
            *p++ = 'A' + tmp - 10;
        } else {
            *p++ = 'a' + tmp - 10;
        }
        u /= base;
    } while (u != 0);

    if (neg_flag) {
        *p++ = '-';
    }

    /* figure out actual length and adjust the maximum length */
    actualLength = p - buf;
    if (length < actualLength) {
        length = actualLength;
    }

    /* add padding */
    if (ladjust) {
        padc = ' ';
    }
    if (neg_flag && !ladjust && (padc == '0')) {
        for (i = actualLength - 1; i < length - 1; i++) {
            buf[i] = padc;
        }
        buf[length - 1] = '-';
    } else {
        for (i = actualLength; i < length; i++) {
            buf[i] = padc;
        }
    }

    /* prepare to reverse the string */
    int begin = 0;
    int end;
    if (ladjust) {
        end = actualLength - 1;
    } else {
        end = length - 1;
    }

    /* adjust the string pointer */
    while (end > begin) {
        char tmp = buf[begin];
        buf[begin] = buf[end];
        buf[end] = tmp;
        begin++;
        end--;
    }

    out(data, buf, length);
}
